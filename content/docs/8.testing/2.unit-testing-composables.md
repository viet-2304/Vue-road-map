# Unit Testing Composables

Composables are functions that use Vue's reactivity — testing them requires understanding how to work with refs and lifecycle hooks outside of components.

## Testing Simple Composables

Composables that don't use lifecycle hooks can be tested directly:

```ts
// composables/useCounter.ts
export function useCounter(initial = 0) {
  const count = ref(initial)
  const increment = () => count.value++
  const decrement = () => count.value--
  const reset = () => count.value = initial
  return { count, increment, decrement, reset }
}
```

```ts
// composables/useCounter.test.ts
import { useCounter } from './useCounter'

describe('useCounter', () => {
  it('starts with initial value', () => {
    const { count } = useCounter(10)
    expect(count.value).toBe(10)
  })

  it('defaults to 0', () => {
    const { count } = useCounter()
    expect(count.value).toBe(0)
  })

  it('increments', () => {
    const { count, increment } = useCounter()
    increment()
    increment()
    expect(count.value).toBe(2)
  })

  it('resets to initial value', () => {
    const { count, increment, reset } = useCounter(5)
    increment()
    increment()
    reset()
    expect(count.value).toBe(5)
  })
})
```

## Testing Computed Properties

```ts
// composables/useFilters.ts
export function useFilters(items: Ref<{ name: string; active: boolean }[]>) {
  const search = ref('')
  const filtered = computed(() =>
    items.value.filter(item =>
      item.active && item.name.toLowerCase().includes(search.value.toLowerCase())
    )
  )
  return { search, filtered }
}
```

```ts
describe('useFilters', () => {
  it('filters by search and active status', () => {
    const items = ref([
      { name: 'Apple', active: true },
      { name: 'Banana', active: false },
      { name: 'Avocado', active: true },
    ])
    const { search, filtered } = useFilters(items)

    expect(filtered.value).toHaveLength(2)

    search.value = 'a'
    expect(filtered.value).toHaveLength(2)  // Apple + Avocado

    search.value = 'app'
    expect(filtered.value).toHaveLength(1)  // Apple only
  })
})
```

## Testing Composables with Lifecycle Hooks

Composables using `onMounted`, `onUnmounted`, etc. need a component context. Use a test helper:

```ts
import { mount } from '@vue/test-utils'
import { defineComponent } from 'vue'

function withSetup<T>(composable: () => T): { result: T; wrapper: any } {
  let result: T
  const wrapper = mount(defineComponent({
    setup() {
      result = composable()
      return {}
    },
    template: '<div></div>',
  }))
  return { result: result!, wrapper }
}
```

```ts
// composables/useTimer.ts
export function useTimer() {
  const elapsed = ref(0)
  let id: number

  onMounted(() => {
    id = window.setInterval(() => elapsed.value++, 1000)
  })

  onUnmounted(() => {
    clearInterval(id)
  })

  return { elapsed }
}
```

```ts
describe('useTimer', () => {
  beforeEach(() => vi.useFakeTimers())
  afterEach(() => vi.useRealTimers())

  it('counts elapsed seconds', () => {
    const { result, wrapper } = withSetup(() => useTimer())

    expect(result.elapsed.value).toBe(0)
    vi.advanceTimersByTime(3000)
    expect(result.elapsed.value).toBe(3)

    wrapper.unmount()  // Triggers cleanup
  })
})
```

## Testing Async Composables

```ts
// composables/useFetchData.ts
export function useFetchData(url: Ref<string>) {
  const data = ref(null)
  const error = ref<string | null>(null)
  const loading = ref(false)

  async function execute() {
    loading.value = true
    try {
      const res = await fetch(url.value)
      data.value = await res.json()
    } catch (e) {
      error.value = (e as Error).message
    } finally {
      loading.value = false
    }
  }

  watch(url, execute, { immediate: true })
  return { data, error, loading }
}
```

```ts
describe('useFetchData', () => {
  it('fetches data successfully', async () => {
    const mockData = { name: 'Alice' }
    global.fetch = vi.fn().mockResolvedValue({
      json: () => Promise.resolve(mockData)
    })

    const url = ref('/api/user')
    const { data, loading } = useFetchData(url)

    await flushPromises()

    expect(data.value).toEqual(mockData)
    expect(loading.value).toBe(false)
  })

  it('handles errors', async () => {
    global.fetch = vi.fn().mockRejectedValue(new Error('Network error'))

    const url = ref('/api/fail')
    const { error } = useFetchData(url)

    await flushPromises()

    expect(error.value).toBe('Network error')
  })
})
```

Use `flushPromises` from `@vue/test-utils` or create your own:

```ts
const flushPromises = () => new Promise(resolve => setTimeout(resolve, 0))
```

## Mocking Dependencies

```ts
// Mock a module
vi.mock('./api', () => ({
  fetchUsers: vi.fn().mockResolvedValue([{ id: 1, name: 'Alice' }])
}))

// Mock localStorage
const mockStorage: Record<string, string> = {}
vi.stubGlobal('localStorage', {
  getItem: (key: string) => mockStorage[key] ?? null,
  setItem: (key: string, value: string) => { mockStorage[key] = value },
  removeItem: (key: string) => { delete mockStorage[key] },
})
```

## Common Pitfalls

| Pitfall | Solution |
|---|---|
| Lifecycle hooks failing outside component | Use `withSetup` helper to provide component context |
| Computed not updating in tests | Computed updates synchronously — just access `.value` |
| Async composable not resolved | `await flushPromises()` or `await nextTick()` |
| Tests leaking state | Reset mocks and refs in `beforeEach` |
