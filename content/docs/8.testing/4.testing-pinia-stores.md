# Testing Pinia Stores

Pinia provides `@pinia/testing` — a dedicated package for creating test-friendly stores with optional stubbing and spying.

## Setup

```bash
npm install -D @pinia/testing
```

## createTestingPinia

`createTestingPinia` creates a Pinia instance where every store is automatically mocked:

```ts
import { createTestingPinia } from '@pinia/testing'
import { mount } from '@vue/test-utils'
import Dashboard from './Dashboard.vue'

const wrapper = mount(Dashboard, {
  global: {
    plugins: [createTestingPinia()],
  },
})
```

### Options

```ts
createTestingPinia({
  createSpy: vi.fn,          // Default in Vitest
  initialState: {             // Pre-set store state
    counter: { count: 10 },
  },
  stubActions: false,         // Set true to auto-stub all actions
})
```

## Testing Store State

```ts
import { setActivePinia, createPinia } from 'pinia'
import { useCounterStore } from './counter'

describe('Counter Store', () => {
  beforeEach(() => {
    setActivePinia(createPinia())
  })

  it('has initial state', () => {
    const store = useCounterStore()
    expect(store.count).toBe(0)
  })

  it('allows direct state mutation in tests', () => {
    const store = useCounterStore()
    store.count = 42
    expect(store.count).toBe(42)
  })

  it('patches multiple properties', () => {
    const store = useCounterStore()
    store.$patch({ count: 5 })
    expect(store.count).toBe(5)
  })
})
```

## Testing Actions

```ts
// stores/todo.ts
export const useTodoStore = defineStore('todo', () => {
  const items = ref<{ id: number; text: string; done: boolean }[]>([])

  function addTodo(text: string) {
    items.value.push({ id: Date.now(), text, done: false })
  }

  function toggleTodo(id: number) {
    const todo = items.value.find(t => t.id === id)
    if (todo) todo.done = !todo.done
  }

  const doneCount = computed(() => items.value.filter(t => t.done).length)

  return { items, addTodo, toggleTodo, doneCount }
})
```

```ts
describe('Todo Store', () => {
  beforeEach(() => {
    setActivePinia(createPinia())
  })

  it('adds a todo', () => {
    const store = useTodoStore()
    store.addTodo('Write tests')

    expect(store.items).toHaveLength(1)
    expect(store.items[0].text).toBe('Write tests')
    expect(store.items[0].done).toBe(false)
  })

  it('toggles completion', () => {
    const store = useTodoStore()
    store.addTodo('Test')
    const id = store.items[0].id

    store.toggleTodo(id)
    expect(store.items[0].done).toBe(true)

    store.toggleTodo(id)
    expect(store.items[0].done).toBe(false)
  })
})
```

## Testing Getters

```ts
it('counts completed todos', () => {
  const store = useTodoStore()
  store.addTodo('A')
  store.addTodo('B')
  store.addTodo('C')

  store.toggleTodo(store.items[0].id)
  store.toggleTodo(store.items[2].id)

  expect(store.doneCount).toBe(2)
})
```

## Testing Async Actions

```ts
// stores/user.ts
export const useUserStore = defineStore('user', () => {
  const user = ref<{ name: string } | null>(null)
  const loading = ref(false)
  const error = ref<string | null>(null)

  async function fetchUser(id: number) {
    loading.value = true
    error.value = null
    try {
      const res = await fetch(`/api/users/${id}`)
      user.value = await res.json()
    } catch (e) {
      error.value = (e as Error).message
    } finally {
      loading.value = false
    }
  }

  return { user, loading, error, fetchUser }
})
```

```ts
describe('User Store', () => {
  beforeEach(() => {
    setActivePinia(createPinia())
  })

  it('fetches user successfully', async () => {
    global.fetch = vi.fn().mockResolvedValue({
      json: () => Promise.resolve({ name: 'Alice' }),
    })

    const store = useUserStore()
    await store.fetchUser(1)

    expect(store.user).toEqual({ name: 'Alice' })
    expect(store.loading).toBe(false)
    expect(store.error).toBeNull()
  })

  it('handles fetch error', async () => {
    global.fetch = vi.fn().mockRejectedValue(new Error('Network error'))

    const store = useUserStore()
    await store.fetchUser(1)

    expect(store.user).toBeNull()
    expect(store.error).toBe('Network error')
  })
})
```

## Testing Stores in Components

```ts
import { createTestingPinia } from '@pinia/testing'
import { useCounterStore } from '@/stores/counter'

it('renders store state', () => {
  const wrapper = mount(CounterDisplay, {
    global: {
      plugins: [
        createTestingPinia({ initialState: { counter: { count: 42 } } }),
      ],
    },
  })

  expect(wrapper.text()).toContain('42')
})

it('calls store action on click', async () => {
  const wrapper = mount(CounterButton, {
    global: {
      plugins: [createTestingPinia({ stubActions: false })],
    },
  })

  const store = useCounterStore()
  await wrapper.find('button').trigger('click')

  expect(store.increment).toHaveBeenCalled()
})
```

## Stubbed vs Real Actions

```ts
// Stubbed — action is replaced with a spy that does nothing
const pinia = createTestingPinia({ stubActions: true })
const store = useCounterStore()
store.increment()           // Does nothing
expect(store.count).toBe(0) // State unchanged
expect(store.increment).toHaveBeenCalled()

// Real — action runs normally
const pinia = createTestingPinia({ stubActions: false })
const store = useCounterStore()
store.increment()
expect(store.count).toBe(1) // State updated
```

## Resetting Store State

```ts
it('resets between tests', () => {
  const store = useCounterStore()
  store.count = 100

  store.$reset()
  expect(store.count).toBe(0)
})
```

For setup-syntax stores, `$reset` needs a plugin or manual implementation. Use `setActivePinia(createPinia())` in `beforeEach` for clean state.

## Common Pitfalls

| Pitfall | Solution |
|---|---|
| Store state leaking between tests | Use `setActivePinia(createPinia())` in `beforeEach` |
| Action not being called | Check `stubActions` setting — stubbed actions are no-ops |
| `$reset` not working with setup stores | Use `beforeEach` to create fresh Pinia instance instead |
| Testing Pinia outside component context | Use `setActivePinia` before calling `useStore()` |
