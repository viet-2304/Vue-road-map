# API Integration

Fetching and managing server data is central to most Vue applications.

## Basic Fetch in Vue

```vue
<script setup>
const users = ref([])
const loading = ref(false)
const error = ref<string | null>(null)

async function fetchUsers() {
  loading.value = true
  error.value = null
  try {
    const res = await fetch('https://api.example.com/users')
    if (!res.ok) throw new Error(`HTTP ${res.status}`)
    users.value = await res.json()
  } catch (e) {
    error.value = (e as Error).message
  } finally {
    loading.value = false
  }
}

onMounted(fetchUsers)
</script>

<template>
  <div v-if="loading">Loading...</div>
  <div v-else-if="error" class="error">{{ error }}</div>
  <ul v-else>
    <li v-for="user in users" :key="user.id">{{ user.name }}</li>
  </ul>
</template>
```

## Composable Pattern

Extract API logic into a reusable composable:

```ts
// composables/useApi.ts
export function useApi<T>(url: string) {
  const data = ref<T | null>(null)
  const error = ref<string | null>(null)
  const loading = ref(false)

  async function execute() {
    loading.value = true
    error.value = null
    try {
      const res = await fetch(url)
      if (!res.ok) throw new Error(`HTTP ${res.status}`)
      data.value = await res.json()
    } catch (e) {
      error.value = (e as Error).message
    } finally {
      loading.value = false
    }
  }

  return { data, error, loading, execute }
}
```

```vue
<script setup>
const { data: users, loading, error, execute } = useApi<User[]>('/api/users')
onMounted(execute)
</script>
```

## Nuxt Data Fetching

Nuxt provides built-in composables that handle SSR, caching, and deduplication:

### useFetch

```vue
<script setup>
const { data: users, status, error, refresh } = await useFetch('/api/users')
// status: 'idle' | 'pending' | 'success' | 'error'
</script>

<template>
  <div v-if="status === 'pending'">Loading...</div>
  <div v-else-if="error">{{ error.message }}</div>
  <ul v-else>
    <li v-for="user in users" :key="user.id">{{ user.name }}</li>
  </ul>
  <button @click="refresh()">Reload</button>
</template>
```

### useAsyncData

For non-fetch async operations:

```vue
<script setup>
const { data: post } = await useAsyncData(
  `post-${route.params.id}`,  // Unique key
  () => queryCollection('posts').where('id', route.params.id).first()
)
</script>
```

### useFetch Options

```vue
<script setup>
const { data } = await useFetch('/api/users', {
  method: 'POST',
  body: { name: 'Alice' },
  headers: { 'Content-Type': 'application/json' },
  query: { page: 1, limit: 10 },

  // Transform response
  transform: (res) => res.data,

  // Caching
  key: 'users-list',

  // Don't fetch on server
  server: false,

  // Lazy — don't block navigation
  lazy: true,

  // Watch reactive sources to re-fetch
  watch: [page],
})
</script>
```

async function createUser(userData: { name: string; email: string }) {

## Error Handling Patterns

### Global Error Handler

```ts
// composables/useApiClient.ts
export function useApiClient() {
  const baseURL = useRuntimeConfig().public.apiBase

  async function request<T>(path: string, options: RequestInit = {}): Promise<T> {
    const res = await fetch(`${baseURL}${path}`, {
      ...options,
      headers: {
        'Content-Type': 'application/json',
        ...options.headers,
      },
    })

    if (!res.ok) {
      if (res.status === 401) navigateTo('/login')
      if (res.status === 403) throw new Error('Forbidden')
      throw new Error(`API Error: ${res.status}`)
    }

    return res.json()
  }

  return { get: (p) => request(p), post: (p, b) => request(p, { method: 'POST', body: JSON.stringify(b) }) }
}
```

## Optimistic Updates

Update UI immediately, rollback on failure:

```ts
async function toggleTodo(todo: Todo) {
  const previousState = todo.done
  todo.done = !todo.done  // Optimistic update

  try {
    await $fetch(`/api/todos/${todo.id}`, {
      method: 'PATCH',
      body: { done: todo.done }
    })
  } catch {
    todo.done = previousState  // Rollback
    toast.show('Failed to update', 'error')
  }
}
```

## Common Pitfalls

| Pitfall | Solution |
|---|---|
| Fetching in `setup()` without await | Use `await useFetch()` or `onMounted` |
| Not handling loading/error states | Always show loading and error UI |
| Duplicate requests | Use Nuxt's built-in deduplication or a cache key |
| Memory leaks from unfinished requests | Use AbortController cleanup |
| CORS errors | Configure server-side proxy or API headers |
