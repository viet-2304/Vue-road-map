# Plugins

Plugins add app-level functionality to Vue — global components, directives, provide/inject, or instance properties.

## Plugin Structure

A plugin is an object with an `install` method, or simply a function:

```ts
// Object form
const myPlugin = {
  install(app, options) {
    // Add global features here
  }
}

// Function form
function myPlugin(app, options) {
  // Add global features here
}

// Usage
app.use(myPlugin, { /* options */ })
```

## What Plugins Can Do

### 1. Register Global Components

```ts
const IconPlugin = {
  install(app) {
    app.component('Icon', IconComponent)
    app.component('Modal', ModalComponent)
  }
}
```

### 2. Register Global Directives

```ts
const DirectivesPlugin = {
  install(app) {
    app.directive('focus', { mounted: (el) => el.focus() })
    app.directive('tooltip', tooltipDirective)
  }
}
```

### 3. Provide App-Level Values

```ts
const ApiPlugin = {
  install(app, options) {
    const apiClient = createApiClient(options.baseURL)
    app.provide('api', apiClient)
  }
}

// In components:
const api = inject('api')
```

### 4. Add Global Properties

```ts
const UtilsPlugin = {
  install(app) {
    app.config.globalProperties.$formatDate = (date: Date) => {
      return date.toLocaleDateString()
    }
  }
}

// In Options API components:
this.$formatDate(new Date())
```

## Practical Example: Toast Plugin

```ts
// plugins/toast.ts
import { ref } from 'vue'

interface Toast {
  id: number
  message: string
  type: 'success' | 'error' | 'info'
}

function createToast() {
  const toasts = ref<Toast[]>([])
  let nextId = 0

  function show(message: string, type: Toast['type'] = 'info') {
    const id = nextId++
    toasts.value.push({ id, message, type })
    setTimeout(() => remove(id), 3000)
  }

  function remove(id: number) {
    toasts.value = toasts.value.filter(t => t.id !== id)
  }

  return { toasts, show, remove }
}

export const ToastPlugin = {
  install(app) {
    const toast = createToast()
    app.provide('toast', toast)
  }
}
```

Usage in components:

```vue
<script setup>
const toast = inject('toast')

function save() {
  try {
    // ... save logic
    toast.show('Saved successfully!', 'success')
  } catch {
    toast.show('Failed to save', 'error')
  }
}
</script>
```

## Plugin Options

Accept configuration when installed:

```ts
interface TranslationOptions {
  locale: string
  messages: Record<string, Record<string, string>>
}

const I18nPlugin = {
  install(app, options: TranslationOptions) {
    const currentLocale = ref(options.locale)

    function t(key: string): string {
      return options.messages[currentLocale.value]?.[key] ?? key
    }

    app.provide('i18n', { t, locale: currentLocale })
    app.config.globalProperties.$t = t
  }
}

// Registration
app.use(I18nPlugin, {
  locale: 'en',
  messages: {
    en: { greeting: 'Hello' },
    vi: { greeting: 'Xin chào' },
  }
})
```

## Plugins in Nuxt

Nuxt has its own plugin system that wraps Vue plugins:

```ts
// plugins/my-plugin.ts
export default defineNuxtPlugin((nuxtApp) => {
  // Access Vue app
  nuxtApp.vueApp.use(SomeVuePlugin)

  // Provide values
  nuxtApp.provide('myService', createService())

  // Access runtime config
  const config = useRuntimeConfig()
})
```

### Plugin Order and Dependencies

```ts
// plugins/01.auth.ts (runs first due to number prefix)
export default defineNuxtPlugin((nuxtApp) => {
  const auth = createAuthService()
  nuxtApp.provide('auth', auth)
})

// plugins/02.api.ts
export default defineNuxtPlugin((nuxtApp) => {
  const auth = nuxtApp.$auth  // Available from previous plugin
  const api = createApiClient(auth)
  nuxtApp.provide('api', api)
})
```

## Common Pitfalls

| Pitfall | Solution |
|---|---|
| Plugin modifying global state | Use `provide/inject` for isolation |
| No TypeScript support for `globalProperties` | Augment `ComponentCustomProperties` interface |
| Plugin order dependencies | Use numbered filenames in Nuxt, or explicit ordering |
| Heavy plugins loaded upfront | Lazy-load with dynamic imports |
